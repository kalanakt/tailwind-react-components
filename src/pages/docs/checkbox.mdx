### Icon Checkbox

<IconCheckbox />

```diff-js IconCheckbox.js
import { useState } from "react";
import { BankIcon } from "@vwjs/icons";

export function IconCheckbox() {
  const [checked, setChecked] = useState(false);
  return (
    <>
      <BankIcon
        onClick={() => setChecked(!checked)}
        width={25}
        height={25}
        fill-rule="nonzero"
        fill={checked ? "blue" : "#d4d4d8"}
      />
    </>
  );
}
```
